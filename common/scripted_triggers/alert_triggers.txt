alert_character_is_worth_blackmailing = {
  AND = { 
    $BLACKMAILER$ = { has_perk = golden_obligations_perk }   # can we demand gold
    $PARTICIPANT$.gold > 9                                   # do they have gold
    $PARTICIPANT$ = { is_imprisoned = no }                   # not in prison
  }                  
}
